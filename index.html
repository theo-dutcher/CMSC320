Part I: Data Processing
This data processing script computes information about the linearly additive subcomponents picked out by Independant Component Analysis (ICA). The first section performs a fast fourier transform to assess which subcomponents have more activity in alpha frequencies (8-12Hz) then in surrounding frequencies (<8Hz and >12Hz). The second section decides which subcomponents to include (and graphs them). The third section considers individual (included) subcomponents and localizes each one. The fourth section converts the coordinates from 'MRI space' to 'voxel space' to allow us to plot results. Together, they make an easily-interpreted dataframe to join the ICA object.

'''
This simply lines up all the subject's trials then applies ICA to
that long list of magnetic sensor readings. That fitted model is
saved in a file for future use.
'''
def decompose(subject, method='fastica'):
    ica_path = ica_dir + '/' + subject + '-ica.fif'
    if not os.path.isfile(ica_path):
        # extracts the data and dimensions of all subject epochs
        epochs = experiment.load_epochs(subject=subject, ndvar=False)['epochs']
        data = epochs._data
        events, sensors, times = data.shape
        
        '''
        To concatenate events we convert the multi-event object with
        dimensions (events x sensors x times) to one epoch object with
        a single event (1 x sensors x times) by multiplying the number
        of time points by the number of events.
        ''' 
        concatenated_data = np.reshape(data, (1, sensors, times * events))
        concatenated_epoch =  mne.EpochsArray(concatenated_data, epochs.info)
        
        # finally we fit the mixing matrix to our concatenated data
        ica = mne.preprocessing.ICA(method=method)
        ica.fit(concatenated_epoch)
        ica.save(ica_path)
    else:
        print('[' + subject + ']: ' + method + ' ICA already exists.')

Section 1: Classification by relative alpha power.
In [7]:
'''
Section I: Classify by relative alpha power.
Here we load the ICA, examine every source (i.e. the activity
of a subcomponent over time with dimensionality (times)) and
perform a fast fourier transform on each. The alpha power is
compared to the power of frequencies below (2-6Hz) and above
(16-20Hz).
'''
def classify(subject, threshold=2):
    epochs = experiment.load_epochs(subject=subject, ndvar=False)['epochs']
    
    # reload ica file and the sources array
    ica_path = ica_dir + '/' + subject + '-ica.fif'
    ica = mne.preprocessing.read_ica(ica_path)
    sources = ica.get_sources(epochs.copy()).get_data()[0]
    n = ica.n_components_

    ratio = []
    for source in sources:
        fft = np.fft.fft(source)
        freq = np.fft.fftfreq(len(source)) * epochs.info['sfreq']
        power = np.square(np.abs(fft))

        alpha = np.mean(np.abs(power[(freq >= 8) * (freq <= 12)]))
        other_up = np.mean(np.abs(power[(freq >= 16) * (freq<= 20)]))
        other_down = np.mean(np.abs(power[(freq >= 2) * (freq <= 6)]))
        other = np.mean([other_up, other_down])

        ratio = ratio + [alpha / other]

    return ratio

'''
Here we take the relative alpha powers and a threshold value
and plot the number of included (high alpha) and excluded sub-
components. We also return a boolean mask for the data.
'''
def threshold(subject, ratio, threshold):
    values = ratio.copy()    
    values.sort()
    values.reverse()
    
    i = 0
    while values[i] > threshold:
        i += 1
    
    # plot included and excluded subcomponents
    ax1 = plt.plot(np.arange(0, i+1, 1), values[:(i+1)], color='b', label='included')
    ax2 = plt.plot(np.arange(i, len(values), 1), values[i:], color='r', label='excluded')
    
    # plot details and line at threshold value
    plt.axhline(y=values[i], xmin=0, xmax = 200, color='k')
    plt.xlim(0, len(values) - 1)
    plt.xlabel('components')
    plt.ylabel('alpha / other')
    plt.title(subject + ' Component Selection')
    plt.legend(loc='upper right')
    plt.show()
    
    # which subcomponents to investigate further
    selected = (np.array(ratio) > threshold).tolist()
    
    return selected

'''
Here we take the selected components and look at the pattern
of activity they represent. Most localize well to single di-
poles, and their locations are stored as well as a goodness-
of-fit metric.
'''
def localize(subject, selected):
    epochs = experiment.load_epochs(subject=subject, ndvar=False)['epochs']
    
    # reload ica file and the components array
    ica_path = ica_dir + '/' + subject + '-ica.fif'
    ica = mne.preprocessing.read_ica(ica_path)
    components = ica.get_components().transpose()
    n = ica.n_components_

    # these lines are responsible for getting relevant head information
    cov = experiment.load_cov()
    sol = mne.read_bem_solution(experiment.get('bem-sol-file'))
    trans = mne.read_trans(experiment.get('trans-file'))
    info = epochs.crop(0.0, 0.005).average().info
    
    x, y, z, fit = [np.nan] * n, [np.nan] * n, [np.nan] * n, [np.nan] * n
    for i, component in enumerate(components):
        if selected[i]:
            # this section transforms the component back into evoked data
            data = [[x] for x in component]
            for bad in epochs.info['bads']:
                n = int(bad[3:])
                data.insert(n-1, [0.0])
            data += [[0.0]] * 3

            # now we make an evoked object for the subcomponent and localize it 
            evoked = mne.EvokedArray(data, info)
            evoked.pick_types(meg=True, eeg=False)
            dipole, _ = mne.fit_dipole(evoked, cov, sol, trans)

            # our dataframe will store each coordinate independently
            pos = dipole.pos.tolist()[0]
            x[i] = pos[0]
            y[i] = pos[1]
            z[i] = pos[2]
            
            # we store a goodness-of-fit metric from the fitting algorithm
            fit[i] = dipole.gof[0]
    return x, y, z, fit

'''
To plot and manupulate positional data, we need to transform
it into voxel space by applying the 4x4 trans file then divid-
ing by the scaling factor (by which subject brains differ from
average brains).
'''
def trans(subject, x, y, z):
    trans = mne.read_trans(experiment.get('trans-file'))['trans']
    
    s_x, s_y, s_z = [], [], []
    for coords in zip(x, y, z):
        scaled_coords = mne.transforms.apply_trans(trans, coords)
        
        # read the cfg file to get the scaling factor (is there a function for this?)
        file = open(mri_dir + '/' + subject + '/' + 'MRI scaling parameters.cfg')
        for line in file:
            match = re.match(r'scale = ([0-9]*\.[0-9]*)', line)
            if match:
                scale = float(match[1])
        
        s_x = s_x + [scaled_coords[0] / scale]
        s_y = s_y + [scaled_coords[1] / scale]
        s_z = s_z + [scaled_coords[2] / scale]
        
    return s_x, s_y, s_z

df_list = []
for subject in subjects:
    df_path = ica_dir + '/' + subject + '-ica-component-data.csv'
    if not os.path.isfile(df_path):
        # section 1
        ratio = classify(subject)
        # section 2
        selected = threshold(subject, ratio, 1.5)
        # section 3
        x, y, z, fit = localize(subject, selected)
        # section 4
        x, y, z = trans(subject, x, y, z)
        # combinine and save the data
        df_frame = {'ratio': ratio,
                    'x': x,
                    'y': y,
                    'z': z,
                    'fit': fit}
        df = pd.DataFrame(df_frame)
        df.to_csv(df_path)
        df_list = df_list + [df]
    else:
        # if we already have the information we replot it.
        df = pd.read_csv(df_path, index_col=0)
        selected = threshold(subject, df['ratio'].tolist(), 1.5)
        df_list = df_list + [df]

Part II: Data Analysis

Here we try to grossly visualize the places in the brain where these alpha sources commonly localize to in different subjects. To do this, we plot all localized dipoles on the same head. Notably, if we split up the results based on whether the alpha ratio is lower (1.5 - 3) or higher (3+) we see that there are two distinct and common loci.

vol = eel.VolumeSourceSpace.from_file(mri_dir, 'fsaverage', src='vol-5')

high_alpha = np.zeros(len(vol.coordinates))
low_alpha = np.zeros(len(vol.coordinates))
for i, row in subcomponent_df.iterrows():
    if row['ratio'] > 1.5:
        point = [row['x'], row['y'], row['z']]
        distances = sp.spatial.distance.cdist(vol.coordinates, [point])
        
        if row['ratio'] > 3:
            high_alpha[np.argmin(distances)] += 1
        else:
            low_alpha[np.argmin(distances)] += 1
            
high_alpha_ndvar = eel.NDVar(high_alpha, dims=vol)
low_alpha_ndvar = eel.NDVar(low_alpha, dims=vol)
total_ndvar = eel.NDVar(low_alpha + high_alpha, dims=vol)
